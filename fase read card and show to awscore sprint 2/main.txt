#include <SPI.h>
#include <MFRC522.h>
#include "secrets.h"
#include "AWS-IoT.h"
#include <WiFi.h>  

// RFID pins
#define SS_PIN  5
#define RST_PIN 22
MFRC522 rfid(SS_PIN, RST_PIN);

// State flags
bool WifiConnected = false;

void SetupWifi() {
  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 60) {
    delay(500);
    Serial.print(".");
    tries++;
  }
  WifiConnected = (WiFi.status() == WL_CONNECTED);
  Serial.println(WifiConnected ? "\nâœ… WiFi connected!" : "\nâŒ WiFi failed");
  if (WifiConnected) {
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
  }
}

void CheckWifi() {
  WifiConnected = (WiFi.status() == WL_CONNECTED);
}

void SetupRFID() {
  SPI.begin();
  rfid.PCD_Init();
  Serial.println("RC522 ready â€” scan a card...");
}

bool ReadUID(String &uidOut) {
  if (!rfid.PICC_IsNewCardPresent()) return false;
  if (!rfid.PICC_ReadCardSerial())   return false;

  uidOut = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    if (rfid.uid.uidByte[i] < 0x10) uidOut += "0";
    uidOut += String(rfid.uid.uidByte[i], HEX);
  }
  uidOut.toUpperCase();

  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();
  return true;
}

void setup() {
  Serial.begin(115200);
  delay(50);

  SetupWifi();
  SetupRFID();

  if (WifiConnected) {
    SetupAWS();
  }
}

void loop() {
  CheckWifi();
  if (WifiConnected) {
    CheckAWS();
  }

  if (WifiConnected && AWSConnected) {
    String uid;
    if (ReadUID(uid)) {
      Serial.print("ðŸ’³ Card UID: ");
      Serial.println(uid);
      PublicaTag(uid);
    }
  }

  delay(50);
}
