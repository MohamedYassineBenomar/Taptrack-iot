	#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN  5
#define RST_PIN 22   // safer pin than GPIO0

MFRC522 rfid(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(115200);
  while (!Serial) delay(10);

  SPI.begin();  // default VSPI pins: SCK=18, MISO=19, MOSI=23, SS=5
  rfid.PCD_Init();
  delay(50);
  Serial.println("RC522 ready. Bring your card close...");
}

void loop() {
  if (!rfid.PICC_IsNewCardPresent()) return;
  if (!rfid.PICC_ReadCardSerial())  return;

  Serial.print("Card UID: ");
  for (byte i = 0; i < rfid.uid.size; i++) {
    Serial.print(rfid.uid.uidByte[i] < 0x10 ? " 0" : " ");
    Serial.print(rfid.uid.uidByte[i], HEX);
  }
  Serial.println();

  rfid.PICC_HaltA();
  rfid.PCD_StopCrypto1();
}
